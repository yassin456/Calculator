    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        int  first=Integer.parseInt(jTextField1.getText());
        int second=Integer.parseInt(jTextField2.getText());
        String  sign1=jTextField3.getText();
        char sign=sign1.charAt(0);
        
            if (sign == '+')
            {
                int result = Add(first,second);
                jTextArea1.setText(String.valueOf(result));
                
                
                jTextField1.setText(" ");
                jTextField2.setText(" ");
                jTextField3.setText(" ");

                
            }
            if (sign == '/')
            {
                double result = Div(first, second);
                   jTextArea1.setText(String.valueOf(result));
                jTextField1.setText("");
                jTextField2.setText("");
                jTextField3.setText("");
            
            }
            if (sign == '*')
            {
                int result = Mul(first, second);
                   jTextArea1.setText(String.valueOf(result));
                jTextField1.setText("");
                jTextField2.setText("");
                jTextField3.setText("");
         
            }
            if (sign == '-')
            {
                int result = Sub(first, second);
                   jTextArea1.setText(String.valueOf(result));
                jTextField1.setText("");
                jTextField2.setText("");
                jTextField3.setText("");
               
            }
        
    }

     public int Add(int first, int second)
        {

            return first + second;

        }
        public int Sub(int first, int second)
        {
            return first - second;
        }
//============================================================
        public int Mul(int first, int second)
        {
            return first * second;
        }
        public Double Div(int first, int second)
        {
            double result = first/second;
            if (second == 0)
            {
                //MessageBox.Show("can not define by zeroo!!!!!");
                return 0.0;

            }
            else
            {

                return result;
            }
        }
